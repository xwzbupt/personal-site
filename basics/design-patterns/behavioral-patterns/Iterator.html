  public boolean hasNext() {
    return cursor < arrayList.size();

  public void next() {
    cursor++;
  }

    if (cursor >= arrayList.size()) {

    checkForComodification();

  public void next() {
    checkForComodification();

    checkForComodification();
    return arrayList.get(cursor);
  }

  }

      return cursor != size;

      if (i >= size)
